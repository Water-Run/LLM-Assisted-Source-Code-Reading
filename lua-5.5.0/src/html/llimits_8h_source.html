80/*
81** 'l_mem' is a signed integer big enough to count the total memory
82** used by Lua. (It is signed due to the use of debt in several
83** computations.) 'lu_mem' is a corresponding unsigned type. Usually,
84** 'ptrdiff_t' should work, but we use 'long' for 16-bit machines.https://gemini.google.com/app/74cb7cf70647ac58
85**
86** 'l_mem' 是一个有符号整数,大到足以计数Lua使用的总内存。
87** (它是有符号的,因为在几个计算中使用了负债/欠款的概念。)
88** 'lu_mem' 是对应的无符号类型。
89** 通常,'ptrdiff_t' 应该可以工作,但我们对16位机器使用 'long'。
90**

92** 1. 如果定义了LUAI_MEM,使用外部定义的类型
93** 2. 如果是32位整数系统(LUAI_IS32INT),使用ptrdiff_t和size_t

100** - 需要表示负数来跟踪内存分配的欠账

113/*
114** MAX_LMEM - l_mem类型的最大值
115**
116** 计算方法详解:
117** 1. cast(lu_mem, 1): 将1转换为无符号内存类型
118** 2. << (l_numbits(l_mem) - 1): 左移到最高位
119** - 对于32位: 1 << 31 = 0x80000000
120** - 对于64位: 1 << 63 = 0x8000000000000000
121** 3. - 1: 减1得到所有低位为1的数
122** - 对于32位: 0x7FFFFFFF (2147483647)
123** - 对于64位: 0x7FFFFFFFFFFFFFFF
124** 4. cast(l_mem, ...): 转换为有符号类型
125**
126** 这个值等于有符号整数的最大值 (INT_MAX或LONG_MAX)
127*/
128#define MAX_LMEM \
129 cast(l_mem, (cast(lu_mem, 1) << (l_numbits(l_mem) - 1)) - 1)

206/*
207** test whether an unsigned value is a power of 2 (or zero)
208** 测试一个无符号值是否是2的幂(或0)
209**
210** 原理:
211** - 2的幂的二进制表示只有一个1: 如 8 = 1000
212** - x-1会将这个1变为0,后面的0变为1: 7 = 0111
213** - x & (x-1) 会得到0
214**
215** 示例:
216** - ispow2(8): 8=1000, 7=0111, 1000&0111=0000, 结果为真
217** - ispow2(6): 6=0110, 5=0101, 0110&0101=0100, 结果为假
218** - ispow2(0): 0=0000, -1=1111, 0000&1111=0000, 结果为真(特殊情况)
219**
220** 用途: 常用于哈希表大小检查,哈希表通常要求大小为2的幂
221*/
222#define ispow2(x) (((x) & ((x) - 1)) == 0)

681/*
682** modulo: defined as 'a - floor(a/b)*b'; the direct computation
683** using this definition has several problems with rounding errors,
684** so it is better to use 'fmod'. 'fmod' gives the result of
685** 'a - trunc(a/b)*b', and therefore must be corrected when
686** 'trunc(a/b) ~= floor(a/b)'. That happens when the division has a
687** non-integer negative result: non-integer result is equivalent to
688** a non-zero remainder 'm'; negative result is equivalent to 'a' and
689** 'b' with different signs, or 'm' and 'b' with different signs
690** (as the result 'm' of 'fmod' has the same sign of 'a').
691**
692** 模运算:定义为'a - floor(a/b)*b';使用这个定义直接计算
693** 会有几个舍入错误的问题,所以最好使用'fmod'。'fmod'给出
694** 'a - trunc(a/b)*b'的结果,因此当'trunc(a/b) ~= floor(a/b)'时
695** 必须进行修正。当除法有非整数负结果时会发生这种情况:
696** 非整数结果等价于非零余数'm';负结果等价于'a'和'b'符号不同,
697** 或'm'和'b'符号不同(因为'fmod'的结果'm'与'a'同号)。
698**
699** luai_nummod - 模运算
700**
701** 实现细节:
702** 1. m = fmod(a, b): 使用C库的fmod
703** - fmod: 返回a - trunc(a/b)*b
704** - 结果符号与a相同
705**
706** 2. 需要修正的情况:
707** - Lua要求: a % b = a - floor(a/b)*b
708** - fmod给出: a - trunc(a/b)*b
709** - 当a/b为负且非整数时,trunc和floor不同
710**
711** 3. 修正条件: ((m > 0) ? (b) < 0 : ((m) < 0 && (b) > 0))
712** - m > 0 且 b < 0: m需要减去|b|,即m += b
713** - m < 0 且 b > 0: m需要加上|b|,即m += b
714**
715** 示例:
716** - 7 % 3: fmod(7,3)=1, 无需修正, 结果=1
717** - -7 % 3: fmod(-7,3)=-1, m<0且b>0, -1+3=2, 结果=2
718** - 7 % -3: fmod(7,-3)=1, m>0且b<0, 1+(-3)=-2, 结果=-2
719*/
720#if !defined(luai_nummod)
721#define luai_nummod(L, a, b, m) \
722 { \
723 (void)L; \
724 (m) = l_mathop(fmod)(a, b); \
725 if (((m) > 0) ? (b) < 0 : ((m) < 0 && (b) > 0)) \
726 (m) += (b); \
727 }
728#endif

773/*
774** lua_numbertointeger converts a float number with an integral value
775** to an integer, or returns 0 if the float is not within the range of
776** a lua_Integer. (The range comparisons are tricky because of
777** rounding. The tests here assume a two-complement representation,
778** where MININTEGER always has an exact representation as a float;
779** MAXINTEGER may not have one, and therefore its conversion to float
780** may have an ill-defined value.)
781**
782** lua_numbertointeger将具有整数值的浮点数转换为整数,
783** 或者如果浮点数不在lua_Integer范围内则返回0。
784** (范围比较很棘手,因为舍入的原因。这里的测试假定补码表示,
785** 其中MININTEGER总是有精确的浮点表示;MAXINTEGER可能没有,
786** 因此它到浮点的转换可能有未定义的值。)
787**
788** 参数:
789** - n: 要转换的浮点数
790** - p: 指向结果整数的指针
791**
792** 返回:
793** - 1: 转换成功,结果存储在*p中
794** - 0: 转换失败(超出范围或不是整数)
795**
796** 范围检查详解:
797** 1. n >= (LUA_NUMBER)(LUA_MININTEGER)
798** - 检查下界
799** - MININTEGER可以精确表示为浮点数
800**
801** 2. n < -(LUA_NUMBER)(LUA_MININTEGER)
802** - 检查上界
803** - 使用-MININTEGER而不是MAXINTEGER
804** - 原因: 补码系统中,|MININTEGER| = MAXINTEGER + 1
805** - -MININTEGER可以精确表示,避免舍入问题
806**
807** 3. *(p) = (LUA_INTEGER)(n)
808** - 执行实际转换
809**
810** 4. , 1
811** - 逗号运算符,返回1表示成功
812**
813** 整体逻辑:
814** - 所有条件都满足时,转换并返回1
815** - 任一条件不满足,返回0
816*/
817#define lua_numbertointeger(n, p) \
818 ((n) >= (LUA_NUMBER)(LUA_MININTEGER) && \
819 (n) < -(LUA_NUMBER)(LUA_MININTEGER) && \
820 (*(p) = (LUA_INTEGER)(n), 1))
