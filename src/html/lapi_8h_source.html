36/*
37** ============================================================================
38** API 检查机制
39** ============================================================================
40** 这部分定义了 api_check 宏,用于在 API 调用时进行断言检查。
41** 根据是否定义了 LUA_USE_APICHECK,有两种不同的实现方式。
42*/
43
44#if defined(LUA_USE_APICHECK)
45/*
46** 【开发模式 - 使用标准 assert】
47** 当定义了 LUA_USE_APICHECK 时,使用 C 标准库的 assert 进行检查。
48** 这种方式在断言失败时会直接终止程序并输出错误信息,适合开发调试。
49**
50** 参数说明:
51** l - lua_State 指针(在这个版本中未使用,但保留参数以保持接口一致)
52** e - 要检查的条件表达式
53** msg - 错误消息(在标准 assert 中未使用)
54*/
55#include <assert.h>
56#define api_check(l, e, msg) assert(e)
57
58#else /* for testing */
59/*
60** 【测试/发布模式 - 使用 Lua 自定义断言】
61** 当未定义 LUA_USE_APICHECK 时,使用 Lua 内部的 lua_assert 进行检查。
62**
63** 宏展开说明:
64** ((void)(l), lua_assert((e) && msg))
65** - (void)(l): 将 l 转为 void 类型,避免"未使用参数"的编译警告
66** - (e) && msg: 条件 e 必须为真,且 msg 提供错误描述信息
67** - lua_assert: Lua 内部定义的断言宏,可能在发布版本中被编译为空操作
68**
69** 使用逗号运算符的技巧:
70** 逗号运算符 (expr1, expr2) 会先执行 expr1,然后执行 expr2,整个表达式的值为 expr2 的值。
71** 这里用来先处理 l 参数(避免警告),然后执行真正的断言检查。
72*/
73#define api_check(l, e, msg) ((void)(l), lua_assert((e) && msg))
74#endif

76/*
77** ============================================================================
78** 栈顶递增宏 - api_incr_top
79** ============================================================================
80** 功能: 将 Lua 栈顶指针向上移动一个位置(压入一个元素后调用)
81**
82** 【详细说明】
83** 这个宏执行两个操作:
84** 1. L->top.p++: 将栈顶指针向上移动一个位置
85** 2. api_check(...): 检查栈是否溢出
86**
87** 【栈溢出检查】
88** 条件: L->top.p <= L->ci->top.p
89** - L->top.p: 当前实际的栈顶位置
90** - L->ci->top.p: 当前调用信息允许的最大栈顶位置
91** - 如果 top.p 超过了 ci->top.p,说明栈空间不足,发生溢出
92**
93** 【使用场景】
94** 每当向 Lua 栈压入一个新值后,都需要调用此宏更新栈顶指针。
95** 例如: lua_pushinteger、lua_pushstring 等函数内部都会使用此宏。
96*/
97/* Increments 'L->top.p', checking for stack overflows */
98/* 递增 'L->top.p',同时检查栈溢出 */
99#define api_incr_top(L) \
100 (L->top.p++, api_check(L, L->top.p <= L->ci->top.p, "stack overflow"))

103** ============================================================================
104** 线程锁定机制宏
105** ============================================================================
106** 功能: 定义进入和离开 Lua 核心代码时执行的宏
107**
108** 【设计目的】
109** 在多线程环境中,这些宏可以被重定义为实际的互斥锁操作,保证线程安全。
110** 在单线程环境中,它们被定义为空操作,不产生任何开销。
111**
112** 【宏定义说明】
113** - lua_lock(L): 在进入 Lua 核心代码前调用(例如开始执行 Lua API 函数)
114** - lua_unlock(L): 在离开 Lua 核心代码后调用(例如 API 函数返回前)
115** - ((void) 0): C 语言中的空语句,不做任何操作
116**
117** 【如何自定义】
118** 在编译时可以通过 -D 选项或在包含此头文件前定义这些宏:
119** #define lua_lock(L) pthread_mutex_lock(&global_mutex)
120** #define lua_unlock(L) pthread_mutex_unlock(&global_mutex)
121**
122** 【注意事项】
123** 如果自定义了这些宏,必须确保:
124** 1. 每个 lua_lock 都有对应的 lua_unlock
125** 2. 不能在持有锁的情况下调用可能阻塞的操作
126** 3. 避免死锁情况
127*/
128/*
129** macros that are executed whenever program enters the Lua core
130** ('lua_lock') and leaves the core ('lua_unlock')
131*/
132/* 宏定义:程序进入 Lua 核心时执行 ('lua_lock'),离开核心时执行 ('lua_unlock') */
133#if !defined(lua_lock)
134#define lua_lock(L) ((void)0)
135#define lua_unlock(L) ((void)0)
136#endif

138/*
139** ============================================================================
140** 调整函数返回结果宏 - adjustresults
141** ============================================================================
142** 功能: 处理函数调用返回多个值时的栈空间调整
143**
144** 【背景知识】
145** Lua 函数可以返回多个值。当被调用的函数返回的值多于调用者预期时,
146** 可能需要调整调用者的栈顶指针,以容纳所有返回值。
147**
148** 【宏参数】
149** - L: lua_State 指针
150** - nres: 期望的返回值数量
151** * 如果 nres >= 0: 期望固定数量的返回值
152** * 如果 nres == LUA_MULTRET: 接受所有返回值(多返回值模式)
153**
154** 【调整逻辑】
155** 条件: (nres) <= LUA_MULTRET && L->ci->top.p < L->top.p
156** 1. (nres) <= LUA_MULTRET: 检查是否是多返回值模式
157** - LUA_MULTRET 通常定义为 -1,表示接受任意数量的返回值
158** - 如果 nres > LUA_MULTRET(即 nres >= 0),不需要调整
159**
160** 2. L->ci->top.p < L->top.p: 实际栈顶超出了当前调用信息的栈顶
161** - L->top.p: 实际的栈顶(包含所有返回值)
162** - L->ci->top.p: 调用信息中记录的栈顶
163** - 如果实际栈顶更高,说明返回值超出预期,需要扩展 ci->top.p
164**
165** 3. L->ci->top.p = L->top.p: 调整调用信息的栈顶以容纳所有返回值
166**
167** 【使用示例】
168** 假设函数 f() 被调用,期望返回 2 个值,但实际返回了 5 个值:
169** - 调用前: ci->top.p 设置为能容纳 2 个返回值
170** - 调用后: top.p 实际移动到了 5 个返回值的位置
171** - adjustresults 将 ci->top.p 调整到 top.p,避免后续操作出错
172*/
173/*
174** If a call returns too many multiple returns, the callee may not have
175** stack space to accommodate all results. In this case, this macro
176** increases its stack space ('L->ci->top.p').
177*/
178/* 如果一个调用返回了太多的多返回值,被调用者可能没有足够的栈空间
179** 来容纳所有结果。在这种情况下,此宏会增加其栈空间 ('L->ci->top.p')。
180*/
181#define adjustresults(L, nres) \
182 { \
183 if ((nres) <= LUA_MULTRET && L->ci->top.p < L->top.p) \
184 L->ci->top.p = L->top.p; \
185 }

187/*
188** ============================================================================
189** 检查栈元素数量宏 - api_checknelems
190** ============================================================================
191** 功能: 确保栈中至少有 n 个元素可用
192**
193** 【详细说明】
194** 这个宏用于在执行栈操作前验证栈中有足够的元素。
195**
196** 【计算逻辑】
197** 条件: (n) < (L->top.p - L->ci->func.p)
198**
199** - L->top.p: 当前栈顶指针(指向下一个可用位置)
200** - L->ci->func.p: 当前函数在栈中的位置
201** - L->top.p - L->ci->func.p: 栈中实际存储的元素数量
202** * func.p 指向当前执行的函数对象
203** * func.p 上方的所有位置都是可用的栈元素
204** * 这个差值就是当前栈中的元素总数
205**
206** - (n) < (元素总数): 检查所需的 n 个元素是否存在
207** * 如果 n >= 元素总数,说明栈中元素不足,触发断言
208**
209** 【使用场景】
210** 在执行需要从栈中获取参数的操作前调用,例如:
211** - lua_tonumber(L, -1): 需要至少 1 个元素
212** - lua_concat(L, 3): 需要至少 3 个元素
213**
214** 【栈布局示意】
215** 低地址 -> 高地址:
216** [...] [func] [arg1] [arg2] [arg3] [空] <- top.p
217** ^ ^
218** func.p top.p
219** |<--- (top.p - func.p) --->|
220** 这个距离就是栈中的元素数量
221*/
222/* Ensure the stack has at least 'n' elements */
223/* 确保栈中至少有 'n' 个元素 */
224#define api_checknelems(L, n) \
225 api_check(L, (n) < (L->top.p - L->ci->func.p), \
226 "not enough elements in the stack")
227
228/*
229** ============================================================================
230** 检查可弹出元素数量宏 - api_checkpop
231** ============================================================================
232** 功能: 确保栈中至少有 n 个元素可以被安全弹出
233**
234** 【详细说明】
235** 这个宏比 api_checknelems 更严格,它不仅检查元素数量,还要考虑
236** to-be-closed (TBC) 变量的影响。
237**
238** 【检查条件】
239** 需要同时满足两个条件:
240**
241** 1. (n) < L->top.p - L->ci->func.p
242** - 与 api_checknelems 相同,确保有足够的元素
243**
244** 2. L->tbclist.p < L->top.p - (n)
245** - L->tbclist.p: to-be-closed 变量列表的指针
246** - L->top.p - (n): 弹出 n 个元素后的栈顶位置
247** - 这个条件确保弹出操作不会影响到需要关闭的变量
248**
249** 【TBC (To-Be-Closed) 变量】
250** Lua 5.4 引入了 <close> 标记,允许变量在离开作用域时自动调用清理函数。
251** 这些变量记录在 tbclist 中。如果弹出操作会移除这些变量,可能导致
252** 资源泄漏或清理函数不被调用。
253**
254** 【为什么需要第二个条件】
255** 考虑以下情况:
256** local x <close> = some_resource() -- x 在 tbclist 中
257** -- 此时如果 API 试图弹出包含 x 的元素,会违反 TBC 约定
258**
259** 第二个条件保证:
260** - tbclist.p < top.p - n: TBC变量的位置要低于弹出后的栈顶
261** - 这样弹出操作不会影响到任何 TBC 变量
262**
263** 【注释中的说明】
264** "Some functions only update a slot after checking it for popping"
265** "某些函数只在检查可以弹出后才更新插槽"
266**
267** 这是一个优化说明:有些函数会先检查能否弹出,然后执行"弹出+压入"
268** 的组合操作。这个检查保证了即使是这种优化场景也是安全的。
269**
270** 【使用场景】
271** - lua_pop(L, n): 弹出 n 个元素
272** - lua_remove(L, index): 移除指定位置的元素
273** - lua_replace(L, index): 替换指定位置的元素(先弹出再压入)
274*/
275/* Ensure the stack has at least 'n' elements to be popped. (Some
276** functions only update a slot after checking it for popping, but that
277** is only an optimization for a pop followed by a push.)
278*/
279/* 确保栈中至少有 'n' 个元素可以被弹出。(某些函数只在检查可以弹出后
280** 才更新插槽,但这只是弹出后紧跟压入操作的一种优化。)
281*/
282#define api_checkpop(L, n) \
283 api_check(L, (n) < L->top.p - L->ci->func.p && L->tbclist.p < L->top.p - (n), \
284 "not enough free elements in the stack")
285
286#endif

300** 1. api_check - API 断言检查
301** - 用途: 在开发阶段捕获 API 误用
302** - 实现: 可配置使用标准 assert 或 Lua 内部断言
303** - 影响: 在发布版本中可以完全禁用

311** - 用途: 多线程环境下的互斥访问
312** - 默认: 空操作(适合单线程)
313** - 自定义: 可重定义为实际的锁机制
314**

316** - 用途: 处理多返回值情况
317** - 时机: 函数调用返回后
318** - 逻辑: 扩展调用者栈空间以容纳所有返回值
319**
320** 5. api_checknelems - 检查元素数量
321** - 用途: 确保有足够的元素可操作
322** - 计算: 基于 top.p 和 func.p 的距离
323** - 场景: 读取栈元素前
324**
325** 6. api_checkpop - 检查可弹出数量
326** - 用途: 确保弹出操作安全
327** - 额外检查: 考虑 TBC 变量的影响
328** - 场景: 弹出或移除栈元素前
329**

331**

333** - 每个可能出错的操作都有对应的检查宏

335** - 在发布阶段可以优化掉检查以提高性能
336**

338** - 优点: 零运行时开销,类型安全,可内联
339** - 技巧: 使用逗号运算符组合多个操作
340** - 注意: 参数可能被多次求值,需要小心副作用

343** - 通过 #if defined 支持不同的编译配置
344** - 开发版本启用所有检查
345** - 发布版本可以移除检查以提高性能
346**
347** 【关键数据结构关系】
348**
349** lua_State (L):
350** ├─ top.p : 栈顶指针(下一个可用位置)
351** ├─ ci : 当前调用信息
352** │ ├─ func.p : 当前函数在栈中的位置
